#### [2246. 相邻字符不同的最长路径](https://leetcode-cn.com/problems/longest-path-with-different-adjacent-characters/)
#树形DP #DFS
~~~c++
/*
i. 树形DP的经典题目
ii. f[u] 定义为从u出发的最长路径(只能往节点方向走)
iii. 直接DFS实现树形DP
*/
class Solution {
public:
    int longestPath(vector<int>& parent, string s) {
        int n = parent.size(); 
        vector<vector<int>> ne(n);
        for (int i = 1; i < n; i ++ ) {
            ne[parent[i]].push_back(i);
        }
        int ans = 0; 
        function<int(int)> dfs = [&](int u) {
            vector<int> child;
            for (auto i: ne[u]) {
                int t = dfs(i);
                if (s[i] != s[u]) child.push_back(t);
            }
            sort(child.begin(), child.end());
            reverse(child.begin(), child.end()); 
            if (child.size() > 1) ans = max(ans, child[0] + child[1] + 1);
            if (child.size() > 0) {
                ans = max(ans, child[0] + 1);
                return child[0] + 1;
            }
            ans = max(ans, 1);
            return 1; 
        };
        dfs(0);
        return ans; 
    }
};
~~~

~~~python
class Solution:
    def longestPath(self, parent: List[int], s: str) -> int:
        ans = 1
        n = len(parent)
        ne = [[] for _ in range(n)]
        for i, v in enumerate(parent):
            if i == 0:
                continue
            ne[v].append(i)
        def dfs(u: int) -> int:
            nonlocal ans
            child = []
            for i in ne[u]:
                t = dfs(i)
                if s[i] != s[u]:
                    child.append(t)
            child.sort(reverse = True)
            if len(child) > 1:
                ans = max(ans, child[0] + child[1] + 1)
            if len(child) > 0:
                ans = max(ans, child[0] + 1)
                return child[0] + 1
            return 1
        dfs(0)
        return ans
~~~

~~~java
class Solution {
    List<List<Integer>> ne;
    char[] cs;
    int ans;
    public int longestPath(int[] parent, String s) {
        int n = parent.length; 
        ne = new ArrayList<>();
        cs = s.toCharArray();
        ans = 1;
        for (int i = 0; i < n; i ++ ) ne.add(new ArrayList<>()); 
        for (int i = 1; i < n; i ++ ) ne.get(parent[i]).add(i);
        dfs(0);
        return ans; 
    }

    public int dfs(int u) {
        List<Integer> child = new ArrayList<>(); 
        for (var i: ne.get(u)) {
            int t = dfs(i);
            if (cs[u] != cs[i]) child.add(t);
        }
        Collections.sort(child, (a, b) -> b - a);
        if (child.size() > 1) {
            ans = Math.max(ans, child.get(0) + child.get(1) + 1);
        } 
        if (child.size() > 0) {
            ans = Math.max(ans, child.get(0) + 1);
            return child.get(0) + 1;
        }
        return 1; 
    }
}
~~~

~~~scala 
import scala.collection.mutable.ArrayBuffer

object Solution {
    val INF = 1 << 30
    def longestPath(parent: Array[Int], s: String): Int = {
        var ans = 1
        val n = parent.size
        val ne = for (i <- 0 until n) yield {
            new ArrayBuffer[Int]()
        }
        for (i <- 1 until n) if(i != 0) {
            ne(parent(i)).append(i)
        }

        def dfs(u: Int): Int = {
            var t1 = -INF
            var t2 = - INF
            for (i <- ne(u)) {
                val t = dfs(i)
                if (s(i) != s(u)) {
                    if (t > t1) {
                        t2 = t1
                        t1 = t
                    } else if (t > t2) {
                        t2 = t
                    }
                }
            }
            if (t2 != - INF) {
                ans = Math.max(ans, t1 + t2 + 1)
            }
            if (t1 != - INF) {
                ans = Math.max(ans, t1 + 1)
                return t1 + 1
            }
            1
        }
        dfs(0)
        ans
    }
}
~~~

~~~go
func min(a int, b int) int {if a > b {return b}; return a}
func max(a int, b int) int {if b > a {return b}; return a}

func longestPath(parent []int, s string) int {
    n := len(parent)
    ne := make([][]int, n)
    for i, v := range parent {
        if i == 0 {
            continue
        }
        ne[v] = append(ne[v], i)
    }
    ans := 1
    var dfs func(int) int
    dfs = func(u int) int {
        child := make([]int, 0)
        for _, v := range ne[u] {
            t := dfs(v)
            if (s[v] != s[u]) {
                child = append(child, t)
            }
        }
        sort.Sort(sort.Reverse(sort.IntSlice(child)))
        if len(child) > 1 {
            ans = max(ans, child[0] + child[1] + 1)
        }
        if len(child) > 0 {
            ans = max(ans, child[0] + 1)
            return child[0] + 1
        }
        return 1
    }
    dfs(0)
    return ans
}
~~~
