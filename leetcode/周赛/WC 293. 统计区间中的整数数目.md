#### [6066. 统计区间中的整数数目](https://leetcode.cn/problems/count-integers-in-intervals/)
#动态开点线段树 #线段树
> 题目大意：
> 实现一种数据结构
> i. 能够添加区间
> ii. 能够查询至少出现在其中一个区间的数字个数

~~~c++
/*
i. 线段树 动态开点
ii. 需要修改，不需要pushdown

*/
const int N = 100010;

struct Node {
    int l, r;
    int cnt;
    int ln, rn; 
} tr[21 * N]; 

int idx; 

void init() {
    idx = 0; 
    tr[++ idx] = {1, (int)1e9, 0, 0, 0};
}

void pushup(int u) {
    tr[u].cnt = tr[tr[u].ln].cnt + tr[tr[u].rn].cnt; 
}

void modify(int u, int l, int r) {
    if (l <= tr[u].l && r >= tr[u].r) { // 当前区间已经完全覆盖，直接修改
        tr[u].cnt = tr[u].r - tr[u].l + 1; 
        return; 
    }

    if (tr[u].r - tr[u].l + 1 == tr[u].cnt) return;   // 当前区间已经完全覆盖，修改不会改变状态

    int mid = tr[u].l + tr[u].r >> 1;
    if (l <= mid) {
	    //需要动态开左子树
        if (tr[u].ln == 0) tr[u].ln = ++ idx, tr[idx] = {tr[u].l, mid, 0, 0, 0};
        modify(tr[u].ln, l, r);
    }
    if (r > mid) {
	    // 需要动态开右子树
        if (tr[u].rn == 0) tr[u].rn = ++ idx, tr[idx] = {mid + 1, tr[u].r, 0, 0, 0}; 
        modify(tr[u].rn, l, r); 
    }
    pushup(u); 
}


class CountIntervals {
public:
    CountIntervals() {
        init();
    }
    
    void add(int left, int right) {
        modify(1, left, right);
    }
    
    int count() {
        return tr[1].cnt; 
    }
};

/**
 * Your CountIntervals object will be instantiated and called as such:
 * CountIntervals* obj = new CountIntervals();
 * obj->add(left,right);
 * int param_2 = obj->count();
 */
~~~

#玛朵莉树 #平衡树
~~~java
/*
i. 主要在于合并
*/
class CountIntervals {
    TreeMap<Integer, Integer> map = new TreeMap<>(); 
    int cnt = 0; 
    public CountIntervals() {
        
    }
    
    public void add(int left, int right) {
        int l = left, r = right; 
        for (var e = map.floorEntry(right); e != null && e.getValue() >= left; e = map.floorEntry(right)) {
            int cl = e.getKey(), cr = e.getValue();
            l = Math.min(l, cl);
            r = Math.max(r, cr);
            cnt -= cr - cl + 1;
            map.remove(cl);
        }
        map.put(l, r);
        cnt += r - l + 1;

    }
    
    public int count() {
        return cnt; 
    }
}

 * Your CountIntervals object will be instantiated and called as such:
 * CountIntervals obj = new CountIntervals();
 * obj.add(left,right);
 * int param_2 = obj.count();
~~~
