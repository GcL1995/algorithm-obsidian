/*
i. 任意一个数都有唯一的二进制表示法
ii. 15! > 1e12所以最多有14个 factorials
iii. 可以枚举factorials的集合(2^14) 然后计算剩余数的二进制位数是1的位数个数
iv. 1和2 是factorials也是powers of two, 为了防止重复在枚举时排除掉这两个数
*/
//

#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <queue>

#define x first
#define y second

using namespace std;
typedef long long LL;
typedef pair<int, int> PII;
typedef pair<double, double> PDD;
typedef unsigned long long ULL;

const int N = 100010;

int t;
LL n;
vector<LL> nums;

int ones(LL x) {
    int res = 0;
    while (x) {
        x = x & (x - 1);
        res ++;
    }
    return res;
}

int main() {
    for (LL i = 4, x = 6; x <= 1e12; x *= i, i ++ ) nums.push_back(x);
    int k = nums.size();
    cin >> t;
    while (t -- ) {
        cin >> n;
        int ans = 1e8;
        for (int i = 0; i < 1 << k; i++) {
            LL x = n;
            for (int j = 0; j < k; j++) {
                if (i >> j & 1) x -= nums[j];
            }
            if (x < 0) continue;
            ans = min(ans, ones(i) + ones(x));
        }
        cout << ans << endl;
    }
    return 0;
}
