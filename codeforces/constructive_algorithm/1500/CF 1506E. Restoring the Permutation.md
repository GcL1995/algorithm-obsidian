/*
i. w[i] != w[i - 1]，那么i位置必为w[i], 同时所有小于w[i]的都可以在后续位置填充
ii. 可以用set或者优先队列维护一个集合，可以填充在后续位置的所有值，如果是需要字典序小则从小的开始，需要的字典序大则从大的开始
*/

#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <set>
#include <queue>

#define x first

using namespace std;
typedef long long LL;
typedef pair<int, int> PII;
typedef pair<double, double> PDD;
typedef unsigned long long ULL;

const int N = 200010;
const int INF = 1e9;

int T, n;
int w[N], g[N];

void solve() {
    cin >> n;

    set<int> S;
    for (int i = 1; i <= n; i ++ ) cin >> w[i];

    for (int i = 1; i <= n; i ++ ) {
        if (w[i] != w[i - 1]) {
            g[i] = w[i];
            for (int j = w[i] - 1; j > w[i - 1]; j -- ) S.insert(j);
        }
        else {
            g[i] = *S.begin();
            S.erase(S.begin());
        }
    }
    for (int i = 1; i <= n; i ++ ) cout << g[i] << ' ';
    cout << endl;

    for (int i = 1; i <= n; i ++ ) {
        if (w[i] != w[i - 1]) {
            g[i] = w[i];
            for (int j = w[i] - 1; j > w[i - 1]; j -- ) S.insert(j);
        }
        else {
            g[i] = *S.rbegin();
            S.erase(*S.rbegin());
        }
    }
    for (int i = 1; i <= n; i ++ ) cout << g[i] << ' ';
    cout << endl;
}

int main() {
    cin >> T;
    while (T --) {
        solve();
    }
    return 0;
}
