/*
i. 令 m = k - (n - k)
观察 m, m + 1, ..., k, k - 1, ... m
同时观察p[m], p[m + 1], ..., p[k], ..., p[m]
可以发现，两者的逆序对数目始终相同
这是因为对于任意的x, y, y, x x < y和 y > x两种情况都会有两个逆序对
而对于x p[k] x 不论x < p[k] 还是 x > p[k] 都会有一个逆序对

ii. 所以p[1], ..., p[m - 1] 必须是1, ..., m - 1
iii.为了字典序最大，p[m] ..., p[k] 为 k, ..., m
*/

    for (int i = 1; i < m; i ++ ) {
        cout << i << ' ';
    }
    for (int i = k; i >= m; i -- ) {
        cout << i << ' ';
    }

}

int main() {

        solve();
    }
