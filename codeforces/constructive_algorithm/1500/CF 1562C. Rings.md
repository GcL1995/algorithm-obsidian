#### [CF 1562C. Rings](https://codeforces.com/problemset/problem/1562/C)
#数学 #构造题 
> 题目大意：
> 给定一个长度为n的0, 1字符串s
> 求得两个s的子串，要求这两个子串长度大于n/2且通过二进制转换得到的数字具有倍数关系
~~~c++
/*
i. 可以前缀0，可以后缀0，那么如果存在0，总可以构造出来一组，前缀0则倍数为1， 后缀0则倍数为2
ii. 如果没有0， 那么全是1，可以构造两个全是1长度为n - 1的数列
*/
//
// Created by Administrator on 2022/4/24.
//
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <queue>

#define x first
#define y second

using namespace std;
typedef long long LL;
typedef pair<int, int> PII;
typedef pair<double, double> PDD;
typedef unsigned long long ULL;

const int N = 200010;
const int INF = 1e9;

int T, n;
char w[N];

void solve() {
    cin >> n;
    memset(w, 0, sizeof w);
    scanf("%s", w + 1);
    for (int i = 1; i <= n; i ++ ) {
        if (w[i] == '0') {
            if (i > n / 2) {
                printf("%d %d %d %d\n", 1, i, 1, i - 1);
                return;
            }
            printf("%d %d %d %d\n", i, n, i + 1, n);
            return;
        }
    }
    printf("%d %d %d %d\n", 1, n - 1, 2, n);
}

int main() {
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
~~~
