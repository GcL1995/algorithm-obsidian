#include <cstring>

typedef long long LL;  
typedef pair<int, int> PII;  
typedef pair<double, double> PDD;  
typedef unsigned long long ULL;  
  
const int N = 200010;  
const int INF = 1e9;  
  
LL T, n;  
int w[N];  
  
void solve() {  
    cin >> n;  
    LL a, b;  
    cin >> a >> b;  
    LL k = n % b;  
    if (a == 1) {  
        if (b == 1 || k == 1) cout << "Yes" << endl;  
        else cout << "No" << endl;  
        return;    }  
    for (LL i = 1; i <= n; i *= a) {  
        if (i % b == k) {  
            cout << "Yes" << endl;  
            return;        }  
    }  
    cout << "NO" << endl;  
}  
  
int main() {  
    cin >> T;  
    while (T--) {  
        solve();  
    }  
    return 0;  
}
