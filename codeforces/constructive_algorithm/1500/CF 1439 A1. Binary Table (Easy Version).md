/*
i. 以4个单元格为单位，可以发现1的数目变化为 4 ---> 1 ---> 2 ---> 3 ----> 0
ii. 按照上述转化方式最多4 * 3 = 12, 4个单元格最多需要12次变化
iii. 如果n， m是奇数，那么需要特别处理边界，也就是窗口滑动从2变1
*/
//

//
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <queue>

#define x first
#define y second

using namespace std;
typedef long long LL;
typedef pair<int, int> PII;
typedef pair<double, double> PDD;
typedef unsigned long long ULL;

const int N = 110;
const int INF = 1e9;

int T, n, m;
char w[N][N];
vector<PII> res;

int count(int x, int y) {
    int ans = 0;
    for (int i = x; i <= x + 1; i ++ ) {
        for (int j = y; j <= y + 1; j ++ ) {
            ans += w[i][j] - '0';
        }
    }
    return ans;
}

void draw(int x, int y) {
    if (count(x, y) == 3) {
        for (int i = x; i <= x + 1; i ++ ) {
            for (int j = y; j <= y + 1; j ++ ) {
                if (w[i][j] == '1') {
                    res.push_back({i, j});
                    w[i][j] = '0';
                }
            }
        }
        return;
    }
    if (count(x, y) == 2) {
        int cnt = 0;
        for (int i = x; i <= x + 1; i ++ ) {
            for (int j = y; j <= y + 1; j ++ ) {
                if (w[i][j] == '1') {
                    if (cnt) continue;
                    res.push_back({i, j});
                    w[i][j] = '0';
                    cnt ++;
                }
                else {
                    res.push_back({i, j});
                    w[i][j] = '1';
                }
            }
        }
        draw(x, y);
        return;
    }
    if (count(x, y) == 1) {
        int cnt = 0;
        for (int i = x; i <= x + 1; i ++ ) {
            for (int j = y; j <= y + 1; j ++ ) {
                if (w[i][j] == '0') {
                    if (cnt >= 2) continue;
                    res.push_back({i, j});
                    w[i][j] = '1';
                    cnt ++;
                }
                else {
                    res.push_back({i, j});
                    w[i][j] = '0';
                }
            }
        }
        draw(x, y);
        return;
    }
    if (count(x, y) == 4) {
        res.push_back({x, y}), res.push_back({x, y + 1}), res.push_back({x + 1, y});
        w[x][y] = '0', w[x][y + 1] = '0', w[x + 1][y] = '0';
        draw(x, y);
    }
}

void solve() {
    cin >> n >> m;
    res.clear();
    memset(w, 0, sizeof w);
    for (int i = 1; i <= n; i ++ ) scanf("%s", w[i] + 1);
    for (int i = 1; i <= n - 1; ) {
        for (int j = 1; j <= m - 1;) {
            draw(i, j);
            if (j == m - 2) j ++;
            else j += 2;
        }
        if (i == n - 2) i ++;
        else i += 2;
    }
    cout << res.size() / 3<< endl;
    for (int i = 0; i < res.size(); i += 3) {
        for (int j = i; j < i + 3; j ++ ) {
            cout << res[j].x << ' ' << res[j].y << ' ';
        }
        cout << endl;
    }
    //for (int i = 1; i <= n; i ++ ) printf("%s\n", w[i] + 1);
}

int main() {
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
