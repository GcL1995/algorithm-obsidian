/*
设 一个人赢了x场，平局y场
那么有 2 * x + y = n - 1
如果 (n - 1) % 2 = 0, x = (n - 1) / 2
否则 x = (n - 2）/ 2

那么可以通过记录一个人赢的和输了的当前场次来构造，遍历如果cnt[i][0] <x, cnt[j][1] < x,那么 i赢了j，否则 i输了j
*/
//
// Created by Administrator on 2022/4/24.
//
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <queue>

#define x first
#define y second

using namespace std;
typedef long long LL;
typedef pair<int, int> PII;
typedef pair<double, double> PDD;
typedef unsigned long long ULL;

const int N = 110;
const int INF = 1e9;

int T, n;
int w[N];
int cnt[N][2];

void solve() {
    cin >> n;
    memset(cnt, 0, sizeof cnt);
    if ((n - 1) % 2 == 0) {
        int x = (n - 1) / 2;
        for (int i = 1; i <= n - 1; i ++ ) {
           for (int j = i + 1; j <= n; j ++ ) {
               if (cnt[i][0] < x && cnt[j][1] < x) {
                   cout << 1 << ' ';
                   cnt[i][0] ++;
                   cnt[j][1] ++;
               }
               else {
                   cout << -1 << ' ';
                   cnt[i][1] ++;
                   cnt[j][0] ++;
               }
           }
        }
    }
    else {
        int x = (n - 2) / 2;
        for (int i = 1; i <= n - 1; i ++ ) {
            for (int j = i + 1; j <= n; j ++ ) {
                if (i % 2 == 1 && j == i + 1) {
                    cout << 0 << ' ';
                }
                else if (cnt[i][0] < x && cnt[j][1] < x) {
                    cout << 1 << ' ';
                    cnt[i][0] ++;
                    cnt[j][1] ++;
                }
                else {
                    cout << -1 << ' ';
                    cnt[j][0] ++;
                    cnt[i][1] ++;
                }
            }
        }
    }
    cout << endl;
}

int main() {
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
