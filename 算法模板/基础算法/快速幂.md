[875. 快速幂](https://www.acwing.com/problem/content/description/877/)
> 题目大意：
> 1≤n≤1000001≤n≤100000,  
	1≤ai,bi,pi≤2×109

$$
\begin{aligned}
&a_i^{b_i} \; mod \; p_i 
\end{aligned}
$$
~~~c++
#include <iostream>

using namespace std; 

typedef long long LL; 

LL qmi(int a, int b, int p) {
    LL res = 1 % p; 
    while (b) {
        if (b & 1) res = res * (LL) a % p; 
        a = a * (LL) a % p; 
        b >>= 1; 
    }
    return res; 
}

int main() {
    int n; 
    cin >> n; 
    
    while (n -- ) {
        int a, b, p; 
        scanf("%d%d%d", &a, &b, &p); 
        printf("%lld\n", qmi(a, b, p)); 
    }
    
    return 0; 
}
~~~
