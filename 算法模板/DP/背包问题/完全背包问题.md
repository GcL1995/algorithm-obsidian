# 完全背包
#完全背包 #DP #背包问题
#### [ACW 3. 完全背包问题](https://www.acwing.com/activity/content/problem/content/998/)
~~~c++
#include <iostream>

using namespace std; 

const int N = 1010; 

int n, m; 
int v[N], w[N]; 
int f[N]; 

int main() {
    cin >> n >> m; 
    
    for (int i = 1; i <= n; i ++ ) cin >> v[i] >> w[i]; 
    
    for (int i = 1; i <= n; i ++ ) {
        for (int j = v[i]; j <= m; j ++ ) {
            f[j] = max(f[j], f[j - v[i]] + w[i]); 
        }
    }
    
    cout << f[m] << endl; 
    
    return 0; 
}
~~~
