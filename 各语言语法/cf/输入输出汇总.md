# python
~~~python
import sys


def main() -> None:
    T = int(input())

    for _ in range(T):
        R, C = list(map(int, sys.stdin.readline().strip().split()))
        dp = [0]*101
        for i in range(R):
            val = list(map(int, sys.stdin.readline().strip().split()))
            for j in range(C):
                if i == 0:
                    if j > 0:
                        dp[j] = dp[j - 1] + val[j]
                    else:
                        dp[j] = val[j]
                elif j == 0:
                    dp[j] = dp[j] + val[j]
                else:
                    dp[j] = max(dp[j], dp[j-1]) + val[j]
        print(dp[C - 1])


if __name__ == '__main__':
    main()
~~~
